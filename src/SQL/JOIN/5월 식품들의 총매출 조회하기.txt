WITH A as (SELECT PRODUCT_ID, SUM(AMOUNT) as AMOUNT
          FROM FOOD_ORDER WHERE PRODUCE_DATE LIKE '2022-05%'
          GROUP BY PRODUCT_ID)
SELECT A.PRODUCT_ID, PRODUCT_NAME, PRICE * A.AMOUNT as TOTAL_SALES
FROM A, FOOD_PRODUCT as B
WHERE A.PRODUCT_ID=B.PRODUCT_ID
ORDER BY 3 DESC, 1;
