WITH A as (SELECT FLAVOR, SUM(TOTAL_ORDER) as TOTAL_ORDER
          FROM FIRST_HALF GROUP BY FLAVOR),
B as (SELECT FLAVOR, SUM(TOTAL_ORDER) as TOTAL_ORDER
          FROM JULY GROUP BY FLAVOR)
SELECT A.FLAVOR
FROM A, B
WHERE A.FLAVOR=B.FLAVOR
ORDER BY A.TOTAL_ORDER + B.TOTAL_ORDER DESC LIMIT 3;
